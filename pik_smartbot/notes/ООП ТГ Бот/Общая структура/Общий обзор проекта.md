class User {
  -id: int
  -fullName: string
  -birthDate: date
  -citizenship: CitizenshipEnum

}

}

  -id: int

  +getInfo(): WorkstationInfo
}

  -id: int

}

  -value: string

  -expiresAt: datetime

  +refresh(): void
}

  -id: int
  -user: User
  -status: string

}

  type

  Permission
